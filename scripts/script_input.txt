STAGE initial
SPEAK "请问你有什么需要帮忙的"
MATCH "我的订单状态如何"
    NEXT ask
MATCH "打个招呼"
    NEXT get-name
DEFAULT 
    NEXT unknown

STAGE get-name
SPEAK "你叫什么名字"
INPUT name
    NEXT hello

STAGE hello
SPEAK "你好" + name
MATCH EMPTY
    NEXT initial

STAGE unknown
SPEAK "听不懂命令"
MATCH EMPTY
    NEXT initial

STAGE ask
SPEAK "一切顺利"
MATCH EMPTY
    NEXT initial
